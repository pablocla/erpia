import { z } from 'zod'

const OLLAMA_BASE_URL = process.env.OLLAMA_URL ?? 'http://localhost:11434'
const DEFAULT_MODEL = process.env.OLLAMA_MODEL ?? 'qwen2.5:14b'
const TIMEOUT_MS = 30000

export interface OllamaMessage {
  role: 'system' | 'user' | 'assistant'
  content: string
}

export interface OllamaResponse {
  model: string
  message: { role: string; content: string }
  done: boolean
}

class OllamaClient {
  private static instance: OllamaClient
  
  static getInstance(): OllamaClient {
    if (!this.instance) this.instance = new OllamaClient()
    return this.instance
  }

  async chat(messages: OllamaMessage[], model = DEFAULT_MODEL): Promise<string> {
    const controller = new AbortController()
    const timeout = setTimeout(() => controller.abort(), TIMEOUT_MS)
    
    try {
      const res = await fetch(`${OLLAMA_BASE_URL}/api/chat`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        signal: controller.signal,
        body: JSON.stringify({ model, messages, stream: false })
      })
      
      if (!res.ok) throw new Error(`Ollama error: ${res.status}`)
      const data: OllamaResponse = await res.json()
      return data.message.content
      
    } catch (err) {
      if ((err as Error).name === 'AbortError') throw new Error('Ollama timeout')
      throw err
    } finally {
      clearTimeout(timeout)
    }
  }

  async chatJSON<T>(
    messages: OllamaMessage[], 
    schema: z.ZodSchema<T>,
    model = DEFAULT_MODEL
  ): Promise<T> {
    const raw = await this.chat(messages, model)
    const clean = raw.replace(/```json|```/g, '').trim()
    try {
      return schema.parse(JSON.parse(clean))
    } catch {
      throw new Error(`IA retornó JSON inválido: ${clean.slice(0, 200)}`)
    }
  }

  async isAvailable(): Promise<boolean> {
    try {
      const res = await fetch(`${OLLAMA_BASE_URL}/api/tags`, { 
        signal: AbortSignal.timeout(3000) 
      })
      return res.ok
    } catch {
      return false
    }
  }
}

export const ollama = OllamaClient.getInstance()

import { prisma } from '@/lib/prisma'

export interface EmpresaContexto {
  empresa: {
    nombre: string
    rubro: string
    localidad: string
    provincia: string
    cuit: string
  }
  snapshot: {
    ventasHoy: { total: number; cantidad: number }
    ventasSemana: { total: number; cantidad: number }
    ventasMes: { total: number; cantidad: number }
    ticketPromedio: number
    stockCritico: Array<{ nombre: string; stock: number; stockMinimo: number; unidad: string }>
    topProductos: Array<{ nombre: string; cantidad: number; total: number }>
    clientesDeudores: Array<{ nombre: string; deuda: number; diasVencido: number }>
    turnosPendientesHoy: number
    turnosPendientesMañana: number
    cajaAbierta: boolean
    saldoCaja: number
  }
  historico: {
    ventasUltimos30Dias: Array<{ fecha: string; total: number; cantidad: number }>
    productosEstancados: Array<{ nombre: string; diasSinVenta: number }>
    clientesInactivos: Array<{ nombre: string; diasSinCompra: number; promedioCompra: number }>
  }
  config: {
    modulosActivos: string[]
    moneda: string
    timezone: string
  }
}

export async function buildEmpresaContexto(empresaId: string): Promise<EmpresaContexto> {
  const ahora = new Date()
  const hoy = new Date(ahora.getFullYear(), ahora.getMonth(), ahora.getDate())
  const semanaAtras = new Date(hoy.getTime() - 7 * 24 * 60 * 60 * 1000)
  const mesAtras = new Date(hoy.getTime() - 30 * 24 * 60 * 60 * 1000)
  const manana = new Date(hoy.getTime() + 24 * 60 * 60 * 1000)

  const [
    empresa,
    ventasHoyAgg,
    ventasSemanaAgg,
    ventasMesAgg,
    stockCritico,
    topProductos,
    clientesDeudores,
    turnosHoy,
    turnosMañana,
    cajaTurno,
    ventasDiarias,
    productosEstancados,
    clientesInactivos
  ] = await Promise.all([
    
    prisma.empresa.findUniqueOrThrow({ 
      where: { id: empresaId },
      select: { nombre: true, rubro: true, localidad: true, provincia: true, cuit: true }
    }),
    
    prisma.venta.aggregate({
      where: { empresaId, createdAt: { gte: hoy }, estado: 'COMPLETADA' },
      _sum: { total: true }, _count: { id: true }
    }),
    
    prisma.venta.aggregate({
      where: { empresaId, createdAt: { gte: semanaAtras }, estado: 'COMPLETADA' },
      _sum: { total: true }, _count: { id: true }
    }),
    
    prisma.venta.aggregate({
      where: { empresaId, createdAt: { gte: mesAtras }, estado: 'COMPLETADA' },
      _sum: { total: true }, _count: { id: true }
    }),

    prisma.producto.findMany({
      where: { 
        empresaId, 
        activo: true,
        deletedAt: null,
        controlaStock: true,
        stockActual: { lte: prisma.producto.fields.stockMinimo }
      },
      select: { nombre: true, stockActual: true, stockMinimo: true, unidadMedida: true },
      take: 15,
      orderBy: { stockActual: 'asc' }
    }),

    prisma.lineaVenta.groupBy({
      by: ['productoId'],
      where: { venta: { empresaId, createdAt: { gte: mesAtras } } },
      _sum: { cantidad: true, subtotal: true },
      orderBy: { _sum: { subtotal: 'desc' } },
      take: 10
    }),

    prisma.cuentaCorrienteCliente.findMany({
      where: { empresaId, saldo: { gt: 0 } },
      include: { cliente: { select: { nombre: true, apellido: true } } },
      orderBy: { saldo: 'desc' },
      take: 10
    }),

    prisma.turno.count({
      where: { empresaId, fecha: { gte: hoy, lt: manana }, estado: { in: ['CONFIRMADO', 'PENDIENTE'] } }
    }),

    prisma.turno.count({
      where: { empresaId, fecha: { gte: manana, lt: new Date(manana.getTime() + 86400000) }, estado: { in: ['CONFIRMADO', 'PENDIENTE'] } }
    }),

    prisma.cajaTurno.findFirst({
      where: { empresaId, estado: 'ABIERTO' },
      select: { saldoActual: true }
    }),

    prisma.$queryRaw<Array<{fecha: string, total: number, cantidad: number}>>`
      SELECT DATE(created_at)::text as fecha, 
             SUM(total)::float as total, 
             COUNT(*)::int as cantidad
      FROM "Venta"
      WHERE empresa_id = ${empresaId} 
        AND created_at >= ${mesAtras}
        AND estado = 'COMPLETADA'
      GROUP BY DATE(created_at)
      ORDER BY fecha DESC
    `,

    prisma.producto.findMany({
      where: {
        empresaId, activo: true, deletedAt: null,
        lineasVenta: { none: { venta: { createdAt: { gte: semanaAtras } } } }
      },
      select: { nombre: true, createdAt: true },
      take: 10
    }),

    prisma.cliente.findMany({
      where: {
        empresaId, activo: true, deletedAt: null,
        ventas: { some: { createdAt: { lt: new Date(Date.now() - 30 * 86400000) } } }
      },
      select: { nombre: true, apellido: true, ventas: { select: { total: true, createdAt: true }, orderBy: { createdAt: 'desc' }, take: 5 } },
      take: 15
    })
  ])

  const totalMes = Number(ventasMesAgg._sum.total ?? 0)
  const cantidadMes = ventasMesAgg._count.id ?? 0

  return {
    empresa: {
      nombre: empresa.nombre,
      rubro: empresa.rubro ?? 'GENERAL',
      localidad: empresa.localidad ?? 'Rosario',
      provincia: empresa.provincia ?? 'Santa Fe',
      cuit: empresa.cuit
    },
    snapshot: {
      ventasHoy: { 
        total: Number(ventasHoyAgg._sum.total ?? 0), 
        cantidad: ventasHoyAgg._count.id ?? 0 
      },
      ventasSemana: { 
        total: Number(ventasSemanaAgg._sum.total ?? 0), 
        cantidad: ventasSemanaAgg._count.id ?? 0 
      },
      ventasMes: { total: totalMes, cantidad: cantidadMes },
      ticketPromedio: cantidadMes > 0 ? Math.round(totalMes / cantidadMes) : 0,
      stockCritico: stockCritico.map(p => ({
        nombre: p.nombre,
        stock: Number(p.stockActual),
        stockMinimo: Number(p.stockMinimo),
        unidad: p.unidadMedida ?? 'u'
      })),
      topProductos: topProductos.map(p => ({
        nombre: p.productoId,
        cantidad: Number(p._sum.cantidad ?? 0),
        total: Number(p._sum.subtotal ?? 0)
      })),
      clientesDeudores: clientesDeudores.map(c => ({
        nombre: `${c.cliente.nombre} ${c.cliente.apellido ?? ''}`.trim(),
        deuda: Number(c.saldo),
        diasVencido: 0
      })),
      turnosPendientesHoy: turnosHoy,
      turnosPendientesMañana: turnosMañana,
      cajaAbierta: !!cajaTurno,
      saldoCaja: Number(cajaTurno?.saldoActual ?? 0)
    },
    historico: {
      ventasUltimos30Dias: ventasDiarias,
      productosEstancados: productosEstancados.map(p => ({
        nombre: p.nombre,
        diasSinVenta: Math.floor((Date.now() - p.createdAt.getTime()) / 86400000)
      })),
      clientesInactivos: clientesInactivos.map(c => {
        const ultimaVenta = c.ventas[0]
        const diasSinCompra = ultimaVenta 
          ? Math.floor((Date.now() - ultimaVenta.createdAt.getTime()) / 86400000) 
          : 999
        const promedioCompra = c.ventas.length > 0 
          ? c.ventas.reduce((a, v) => a + Number(v.total), 0) / c.ventas.length 
          : 0
        return { 
          nombre: `${c.nombre} ${c.apellido ?? ''}`.trim(), 
          diasSinCompra, 
          promedioCompra: Math.round(promedioCompra) 
        }
      })
    },
    config: {
      modulosActivos: [],
      moneda: 'ARS',
      timezone: 'America/Argentina/Buenos_Aires'
    }
  }
}

```typescript
import { ollama } from './ollama-client'
import { buildEmpresaContexto, EmpresaContexto } from './context-builder'
import { buildSystemPrompt } from './system-prompts'
import { z } from 'zod'
import { prisma } from '@/lib/prisma'

// Schemas Zod para parsear respuestas de la IA
const AlertaSchema = z.object({
  alertas: z.array(z.object({
    tipo: z.enum(['stock', 'cobranza', 'cliente_inactivo', 'turno', 'caja', 'venta', 'general']),

const ReporteSchema = z.object({

const MensajeWhatsAppSchema = z.object({

const ProyeccionSchema = z.object({

  proyeccion_mes: z.number(),
  confianza: z.enum(['alta', 'media', 'baja']),
  factores: z.array(z.string()),

    cantidad_sugerida: z.number(),

  }))

const ClasificacionProductoSchema = z.object({
  categoria: z.string(),
  subcategoria: z.string().optional(),
  alicuota_iva: z.enum(['0', '10.5', '21', '27']),

  precio_sugerido: z.number().optional(),

export class AIService {
  private static instance: AIService
  
  static getInstance(): AIService {
    if (!this.instance) this.instance = new AIService()
    return this.instance
  }

  // ─────────────────────────────────────────────
  // MÉTODO PRINCIPAL: Chat libre con contexto del negocio
  // ─────────────────────────────────────────────
  async chat(empresaId: string, userMessage: string, historial: Array<{role: 'user'|'assistant', content: string}> = []): Promise<string> {
    const available = await ollama.isAvailable()
    if (!available) return 'El asistente IA no está disponible en este momento. Verificá que Ollama esté corriendo.'
    
    try {
      const contexto = await buildEmpresaContexto(empresaId)
      const systemPrompt = buildSystemPrompt(contexto)
      
      const messages = [
        { role: 'system' as const, content: systemPrompt },
        ...historial,
        { role: 'user' as const, content: userMessage }
      ]
      
      return await ollama.chat(messages)
    } catch (err) {
      console.error('[AIService.chat]', err)
      return 'No pude procesar tu consulta. Intentá de nuevo.'
    }
  }

  // ─────────────────────────────────────────────
  // ALERTAS INTELIGENTES AUTOMÁTICAS
  // ─────────────────────────────────────────────
  async generarAlertas(empresaId: string): Promise<z.infer<typeof AlertaSchema>> {
    try {
      const contexto = await buildEmpresaContexto(empresaId)
      
      const messages = [
        { 
          role: 'system' as const, 
          content: buildSystemPrompt(contexto) + '\n\nRespondé ÚNICAMENTE con JSON válido, sin texto adicional ni markdown.'
        },
        { 
          role: 'user' as const, 
          content: `Analizá todos los datos del negocio y generá las alertas más importantes para hoy.
          
          Considerá:
          1. Stock bajo en productos críticos para el rubro
          2. Clientes con deuda vencida hace más de 30 días
          3. Clientes inactivos que solían comprar con frecuencia

# PASO 2 — API ROUTES

## 2.1 Creá /app/api/ai/chat/route.ts

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { aiService } from '@/lib/ai/ai-service'
import { z } from 'zod'
import { getServerSession } from 'next-auth'
import { authOptions } from '@/lib/auth/auth-options'

const ChatRequestSchema = z.object({
  mensaje: z.string().min(1).max(2000),
  historial: z.array(z.object({
    role: z.enum(['user', 'assistant']),
    content: z.string()
  })).optional().default([])
})

export async function POST(req: NextRequest) {
  try {
    const session = await getServerSession(authOptions)
    if (!session?.user?.empresaId) return NextResponse.json({ error: 'No autorizado' }, { status: 401 })
    
    const body = ChatRequestSchema.parse(await req.json())
    const respuesta = await aiService.chat(session.user.empresaId, body.mensaje, body.historial)
    
    return NextResponse.json({ success: true, data: { respuesta } })
  } catch (err) {
    return NextResponse.json({ success: false, error: (err as Error).message }, { status: 500 })
  }
}

Lógica:
1. Obtener todas las empresas activas con módulo IA habilitado
2. Para cada empresa:
   a. buildEmpresaContexto()
   b. aiService.generarAlertas() → guardar en tabla AlertaIA

   d. Si es lunes → aiService.generarReporte('semana') → guardar en ReporteIA
