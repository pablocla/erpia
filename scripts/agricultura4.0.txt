```prisma
// ── SENSORES DE CAMPO ──────────────────────────────────────────

model Sensor {
  id          Int      @id @default(autoincrement())
  nombre      String   // "Sensor Lote Norte A"
  tipo        SensorTipo
  deviceEUI   String   @unique  // LoRaWAN device identifier
  loteId      Int
  lote        Lote     @relation(fields: [loteId], references: [id])
  activo      Boolean  @default(true)
  lat         Float?
  lon         Float?
  empresaId   Int
  lecturas    SensorLectura[]
  createdAt   DateTime @default(now())
}

enum SensorTipo {
  HUMEDAD_SUELO
  TEMPERATURA_SUELO
  TEMPERATURA_AIRE
  CAUDAL_RIEGO
  PRESION_AGUA
  LLUVIA
  VIENTO
}

model SensorLectura {
  id        BigInt   @id @default(autoincrement())
  sensorId  Int
  sensor    Sensor   @relation(fields: [sensorId], references: [id])
  valor     Float
  unidad    String   // "%" | "°C" | "mm" | "L/min" | "hPa"
  timestamp DateTime @default(now())
  empresaId Int
  // Para TimescaleDB: convertir en hypertable por timestamp
  @@index([sensorId, timestamp])
}

// ── RIEGO DE PRECISIÓN ─────────────────────────────────────────

model ZonaRiego {
  id          Int      @id @default(autoincrement())
  nombre      String   // "Zona A — Pivote 1"
  loteId      Int
  lote        Lote     @relation(fields: [loteId], references: [id])
  geoJson     Json?    // polígono de la zona dentro del lote
  valvulaId   String?  // ID físico del actuador/válvula IoT
  tipoRiego   String   // "GOTEO" | "ASPERSION" | "PIVOTE"
  caudal      Float?   // L/hora de diseño
  activa      Boolean  @default(false)
  empresaId   Int
  eventos     EventoRiego[]
  programas   ProgramaRiego[]
}

model EventoRiego {
  id           Int       @id @default(autoincrement())
  zonaId       Int
  zona         ZonaRiego @relation(fields: [zonaId], references: [id])
  inicio       DateTime
  fin          DateTime?
  duracionMin  Int?      // calculado al cerrar
  volumenLitros Float?
  trigger      RiegoTrigger @default(MANUAL)
  operadorId   Int?
  etoCultivo   Float?    // mm/día de referencia
  observaciones String?
  empresaId    Int
}

enum RiegoTrigger {
  MANUAL
  PROGRAMADO
  SENSOR       // activado por umbral de humedad de suelo
  IA           // recomendado por modelo AquaCrop/Claude
}

model ProgramaRiego {
  id          Int       @id @default(autoincrement())
  zonaId      Int
  zona        ZonaRiego @relation(fields: [zonaId], references: [id])
  nombre      String
  diaSemana   Int[]     // 0=Dom, 1=Lun...6=Sab
  horaInicio  String    // "06:00"
  duracionMin Int
  activo      Boolean   @default(true)
  empresaId   Int
}

// ── TELEMETRÍA DE MAQUINARIA ───────────────────────────────────

model Maquina {
  id           Int      @id @default(autoincrement())
  nombre       String   // "John Deere 8R 310"
  marca        MaquinaMarca
  modeloNombre String?
  serialNumber String?
  apiMachineId String?  // ID en John Deere Ops Center o AGCO Connect
  empresaId    Int
  logs         MaquinaLog[]
}

enum MaquinaMarca {
  JOHN_DEERE
  AGCO
  CASE_IH
  NEW_HOLLAND
  PAUNY        // fabricante argentino
  APACHE       // fabricante argentino
  OTRO
}

model MaquinaLog {
  id          BigInt   @id @default(autoincrement())
  maquinaId   Int
  maquina     Maquina  @relation(fields: [maquinaId], references: [id])
  timestamp   DateTime @default(now())
  lat         Float?
  lon         Float?
  velocidad   Float?   // km/h
  operacion   String?  // "SIEMBRA" | "COSECHA" | "PULVERIZACION" | "TRANSPORTE"
  loteId      Int?
  horasMotor  Float?
  combustible Float?   // litros consumidos
  empresaId   Int
  @@index([maquinaId, timestamp])
}

// ── NDVI Y DATOS SATELITALES ───────────────────────────────────

model NdviRegistro {
  id         Int      @id @default(autoincrement())
  loteId     Int
  lote       Lote     @relation(fields: [loteId], references: [id])
  fecha      DateTime @db.Date
  ndviMin    Float
  ndviMax    Float
  ndviMedio  Float
  fuente     String   @default("SENTINEL2") // "SENTINEL2" | "MODIS" | "ESTIMADO"
  imageUrl   String?  // URL de la imagen procesada
  empresaId  Int
  @@unique([loteId, fecha])
}

// ── RECOMENDACIONES IA ─────────────────────────────────────────

model RecomendacionIA {
  id                 Int      @id @default(autoincrement())
  loteId             Int
  zonaRiegoId        Int?
  debeRegar          Boolean
  volumenRecomendado Float?   // litros
  razon              String   // texto del modelo
  proximoCheckHoras  Int      @default(6)
  inputData          Json     // snapshot de lo que se mandó a Claude
  createdAt          DateTime @default(now())
  empresaId          Int
}
```

## FASE B — API ROUTES NUEVAS

Creá estas rutas siguiendo el patrón existente de `/api/agro/`.

### `/api/agro/sensores/ingest/route.ts`
```
POST — recibe payload de Node-RED/MQTT bridge:
{
  deviceEUI: string,
  valor: number,
  unidad: string,
  timestamp: string (ISO)
}
→ busca el Sensor por deviceEUI
→ inserta SensorLectura en Prisma
→ si tipo es HUMEDAD_SUELO y valor < umbral configurado → disparar lógica de alerta
→ responde 200 OK { sensorId, lecturaId }
```

### `/api/agro/sensores/[sensorId]/lecturas/route.ts`
```
GET — últimas N lecturas de un sensor
Query params: hours=24 (default), limit=100
→ consulta SensorLectura con índice timestamp DESC
→ responde array para Recharts LineChart
```

### `/api/agro/riego/[zonaId]/trigger/route.ts`
```
POST — dispara o detiene riego manual
Body: { accion: "INICIAR" | "DETENER", operadorId: number }
→ si INICIAR: crea EventoRiego con inicio=now(), trigger=MANUAL, actualiza ZonaRiego.activa=true
→ si DETENER: cierra el EventoRiego abierto, calcula duracionMin y volumenLitros estimado
→ aquí irá la llamada al actuador IoT real (stub por ahora — log + HTTP call al gateway)
```

### `/api/agro/riego/[zonaId]/programa/route.ts`
```
GET — lista programas de riego de la zona
POST — crea nuevo programa
PUT — actualiza programa (cambiar días/hora/duración/activo)
DELETE — elimina programa
```

### `/api/agro/maquinas/sync/route.ts`
```
POST — sincroniza telemetría de maquinaria externa
Lógica:
  1. Para cada Maquina con marca=JOHN_DEERE y apiMachineId:
     → GET https://sandboxapi.deere.com/platform/machines/{id}/locations
       (John Deere Operations Center API — OAuth 2.0)
     → insertar MaquinaLog con coordenadas, operacion, horasMotor
  2. Para cada Maquina con marca=AGCO:
     → GET https://api.agconet.com/v1/assets/{id}/telemetry
     → mismo proceso
  3. Para fabricantes argentinos (PAUNY, APACHE):
     → stub: loguear "Sin API disponible — integración manual"
  Retorna: { sincronizadas: N, errores: [] }
```

### `/api/agro/ndvi/[loteId]/route.ts`
```
GET — obtiene NDVI actual del lote
Lógica:
  1. Buscar Lote por id, extraer geoJson (polígono)
  2. Si existe NdviRegistro reciente (< 5 días) → retornar del cache
  3. Si no: llamar Sentinel Hub EvalScript API:
     POST https://services.sentinel-hub.com/api/v1/process
     evalscript: NDVI = (B08 - B04) / (B08 + B04)
     bbox: bounding box del geoJson del lote
  4. Guardar NdviRegistro → retornar { ndviMedio, ndviMin, ndviMax, fecha, imageUrl }
  5. Fallback: si Sentinel falla → retornar último NdviRegistro conocido con flag "cached"
```

### `/api/agro/clima/[loteId]/route.ts`
```
GET — pronóstico 16 días para el lote
Lógica:
  1. Buscar coordenadas centroide del lote (de geoJson o de campos lat/lon)
  2. Llamar Open-Meteo (gratis, sin API key):
     GET https://api.open-meteo.com/v1/forecast
       ?latitude={lat}&longitude={lon}
       &daily=precipitation_sum,et0_fao_evapotranspiration,
              temperature_2m_max,temperature_2m_min,soil_temperature_0cm
       &timezone=America/Argentina/Buenos_Aires
       &forecast_days=16
  3. Detectar alertas: helada si temp_min < 0, lluvia > 30mm, viento > 60km/h
  4. Cachear 1h en memory cache (no llamar en cada render)
  5. Retornar { forecast: [], alertas: [], etoDiario: number }
```

### `/api/agro/ia/recomendar-riego/route.ts`
```
POST — recomendación de riego con Claude
Body: { loteId, zonaRiegoId }
Lógica:
  1. Fetch últimas 24h de SensorLectura tipo HUMEDAD_SUELO de la zona
  2. Fetch pronóstico Open-Meteo (reusar /api/agro/clima/[loteId])
  3. Fetch NdviRegistro más reciente del lote
  4. Buscar Lote.cultivoActual para ajustar parámetros
  5. Llamar Anthropic API (claude-sonnet-4-20250514):
     System: "Sos un agrónomo experto en riego de precisión para cultivos argentinos..."
     User: JSON con { humedadSuelo: [], pronostico: {}, ndvi: {}, cultivo: string }
     → Respuesta estructurada JSON:
       { debeRegar: boolean, volumenLitros: number, razon: string, 
         proximoCheckHoras: number, riesgoHelada: boolean, alertas: string[] }
  6. Guardar RecomendacionIA en Prisma
  7. Retornar la recomendación al cliente
```

## FASE F — BRIDGE IOT (Node-RED → Next.js)

Generá el JSON de un flujo Node-RED que:
1. Se subscribe a MQTT topic: `erp/sensores/{empresaId}/{loteId}/{sensorId}`
2. Parsea payload: `{ valor: number, unidad: string, timestamp: string }`
3. Enriquece con deviceEUI del mensaje MQTT
4. POST a `https://{ERP_HOST}/api/agro/sensores/ingest` con Bearer token
5. En caso de error HTTP: reintenta 3 veces con backoff exponencial
6. Log de cada ingesta en debug node

Para sensores LoRaWAN en Argentina:
- Red disponible: WND Argentina (LoRaWAN público), IOT Factory, Grandata
- Alternativa campo sin cobertura: LoRa punto a punto con gateway propio
- Gateway recomendado open source: RAK7244 o Dragino LG308
